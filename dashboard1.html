function doGet(e) { const ss = SpreadsheetApp.openById('1go8pzPAR3FCuQVULSfnZjoMhRkektjwFmyAkqYCmEVM'); const name = e.parameter.sheet; const sheet = name ? ss.getSheetByName(name) : ss.getSheets()[0]; const out = sheet ? 🗄️ ตั้งค่า URL สำหรับบันทึกฐานข้อมูล (ทำครั้งเดียว)

ใช้ Apps Script เดิม หรือ Deploy ใหม่ โดยเพิ่ม doPost ด้านล่างนี้ลงใน script เดิม:

function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.openById('1go8pzPAR3FCuQVULSfnZjoMhRkektjwFmyAkqYCmEVM'); let sheet = ss.getSheetByName(payload.sheet); if (!sheet) sheet = ss.insertSheet(payload.sheet); const rows = payload.rows; if (rows && rows.length > 0) { const maxCols = Math.max(...rows.map(r => r.length)); const normalized = rows.map(r => { while (r.length < maxCols) r.push(''); return r; }); sheet.getRange(sheet.getLastRow() + 1, 1, normalized.length, maxCols).setValues(normalized); } return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false, error:err.message})) .setMimeType(ContentService.MimeType.JSON); } }
